\startenvironment env_TikZ_Disabled

% This environment uses already existing PDFs rather than making new TikZ diagrams.

% The \marginTikZ command has three arguments.
% #1 Usually a vertical space command, but could be anything
% #2 PDF's name, also used as reference name
% #3 Caption contents

\define[3]\marginTikZ{
  \startplacefigure[location=margin, reference=fig:#2, title={#3}]
    #1\externalfigure[#2][width=\rightmarginwidth]%
  \stopplacefigure
}

\stopenvironment
